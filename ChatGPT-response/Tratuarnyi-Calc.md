from dataclasses import dataclass

@dataclass

    name: str
    sizemm: str                     # "477*167*60"
    klassbetona: str
    amount_poddon_m2: float         # –º¬≤ –Ω–∞ –ø–æ–¥–¥–æ–Ω–µ
    massa_poddona_kg: int           # –º–∞—Å—Å–∞ –ø–æ–ª–Ω–æ–≥–æ –ø–æ–¥–¥–æ–Ω–∞
    price_grey_kzt: int             # —Ü–µ–Ω–∞ –º¬≤ (—Å–µ—Ä—ã–π)

    def calc_area_m2(self):

        length_mm = float(length_mm)
        width_mm = float(width_mm)
        return (length_mm / 1000) * (width_mm / 1000)

    # ---- 2. –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–ª–∏—Ç–æ–∫ –Ω–∞ –ø–æ–¥–¥–æ–Ω ----

        if color == "grey":
            return area_one * self.price_grey_kzt

    # ---- 4. –í–µ—Å –æ–¥–Ω–æ–π –ø–ª–∏—Ç–∫–∏ ----

        return self.massa_poddona_kg / qty

    # ---- 5. –°—Ç–æ–∏–º–æ—Å—Ç—å –ø–æ–¥–¥–æ–Ω–∞ ----
    def calc_pallet_price(self, color="grey"):
        if color == "grey":

        elif color == "color":
            return self.amount_poddon_m2 * self.price_color_kzt
